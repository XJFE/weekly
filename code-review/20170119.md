#### 关于Vue的组件以及页面的一些写法
+ 关于如何写好一个请求:
```javascript
// 获取用户的信息
userInfo() {
    let query = {
        pid: this.$route.params.id
        // TODO 其它的参数
    };
    let interfaceUrl = Vue.ClientUrl.user_info_list;
    // 发送请求
    Vue.ClientHttp(this).GET(query, interfaceUrl)
        .then((res) => {
            // [code = 10000]  
            // 如果使用ES6的箭头函数这里的this就是外部的this值表示当前的vue实例
            // 避免使用多余的变量来捕获外部的this值
            if(res.code === 10000) {
                // TODO
            }
            else {
                // TODO 捕获错误，进一步处理错误
            }
        })
}
```
